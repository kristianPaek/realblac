SELECT  package.icon, album.cat, members_data.postcode, files.featured, members.active AS ThisApproved, members.msgStatus, members.video_duration, files.bigimage, files.type, files.approved, files.adult_content, members_online.logid AS onlinenow, members.id, members.packageid, members.username, members.highlight, members.lastlogin, members_data.gender , members_data.headline, members_data.description, members_data.age, members_data.location, members_data.country 

        FROM members

        LEFT JOIN members_mymatch ON (members.id = members_mymatch.user_id ) 

        INNER JOIN members_data ON (  members.id = members_data.uid ) LEFT JOIN members_online ON ( members_online.logid = members.id )  LEFT JOIN package ON ( members.packageid = package.pid )  LEFT JOIN files ON ( files.uid = members.id AND files.default LIKE '%1%' AND files.type='photo' AND title != 'Message Photo')   LEFT JOIN album ON ( album.aid = files.aid) WHERE  members.active ='active' AND activate_code='OK' AND (members_mymatch.race LIKE '%214%' OR members_mymatch.race = '') AND  (members_mymatch.min_height <= '84' OR members_mymatch.min_height = '') AND  (members_mymatch.max_height > '84' OR members_mymatch.max_height = '') AND  (members_mymatch.education LIKE '%5969%' OR members_mymatch.education = '') AND  (members_mymatch.body_type LIKE '%5762%' OR members_mymatch.body_type = '') AND  (members_mymatch.kids LIKE '%5716%' OR members_mymatch.kids = '') AND  (members_mymatch.min_age <= 47 AND members_mymatch.max_age > 47) AND  members.email !='' AND members.visible = 'yes' AND members_data.gender='63' AND (members_data.em_m5z20131006='5776' OR members_data.em_m5z20131006 is NULL)  GROUP BY members.id ORDER BY members.lastlogin DESC LIMIT 570,30
