		FROM members
		INNER JOIN members_data ON (  members.id = members_data.uid )INNER JOIN members_network ON ( members.id = members_network.to_uid AND members_network.uid='27597'  AND members_network.type= ( '3' ) )WHERE  members.active ='active' AND  (members.packageid !='68' and members.packageid !='70' and members.packageid !='19') AND activate_code='OK' AND members.email !='' AND members.visible = 'yes' AND members_data.gender='63' AND (members_data.em_m5z20131006='5776' OR members_data.em_m5z20131006 is NULL)  GROUP BY members.id ORDER BY members_network.approved DESC, members_network.id DESC  LIMIT 0,30
